// currying
// 함수의 합성이 과연 좋은 방법일까?
function currying(func){
    return function(firstValue){
        return function(secondValue) {
            return func(firstValue, secondValue);
        }
    }
}
 
const add = currying((a,b)=>{ return a+b});
